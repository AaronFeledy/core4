# Drupal

Spin up a Drupal 11 stack with PHP, MariaDB, Drush, and Composer. After the app
starts, use `lando drupal-scaffold` to install the pinned Drupal 11 project and
project-local Drush through a retryable stage-then-commit workflow. The walkthrough
below runs that scaffold, verifies its outputs, checks Drush, and tears the app down.

<Guide>
  <Scenario id="reader-path" layer="e2e" tags={["@smoke"]} render>
    <Step name="scaffold">
      <Variable name="appName" value="drupal" display="my-drupal-app" />
      <Run command="lando init --recipe drupal --name={{appName}} --yes" />
    </Step>
    <Step name="start">
      <Run command="lando start" />
      <Verify event="post-start" />
    </Step>
    <Step name="scaffold-drupal">
      <Run command="lando drupal-scaffold" />
      <Verify command="lando exec appserver -- test -f /app/composer.json" />
      <Verify command="lando exec appserver -- test -x /app/vendor/bin/drush" />
    </Step>
    <Step name="check-drush">
      <Run command="lando drush --version" />
    </Step>
    <Step name="cleanup">
      <Cleanup />
      <Run command="lando destroy -y" />
    </Step>
  </Scenario>
</Guide>
