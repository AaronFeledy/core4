    <Step name="answers">
      <Variable name="answer" value="--answer key=value" display="Repeatable; supplies a recipe prompt answer so prompts that take defaults can be auto-answered" />
      <Variable name="option" value="--option key=value" display="Synonym for --answer; on a key collision --option wins because answers are merged before options (last write wins)" />
      <Variable name="yes" value="--yes" display="Accepts every recipe prompt's default without asking; the -y shorthand is equivalent" />
      <Variable name="noInteractive" value="--no-interactive" display="Never prompts; remaining prompts must be satisfied by defaults or --answer, otherwise the recipe fails; --non-interactive is accepted as a synonym" />
    </Step>
  </Scenario>
