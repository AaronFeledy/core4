# Drive Lando through MCP

An MCP client starts `lando mcp` as a child process and owns its stdin and stdout
pipes. The server exchanges one JSON-RPC object per line and runs until the client
closes stdin. Serve mode writes protocol frames only; it does not wrap startup or
shutdown in a Lando command-result envelope.

      <Run
        runtime="library"
        code={`const { makeStdioMcpTransport } = yield* Effect.promise(() => import("../../../../../core/src/mcp/stdio-transport.ts"));
const { buildCatalog, computeEffectiveAllowlist } = yield* Effect.promise(() => import("../../../../../core/src/mcp/catalog.ts"));
const { mcpRegistryFromCompiled } = yield* Effect.promise(() => import("../../../../../core/src/cli/commands/meta/mcp.ts"));
const { default: compiledCommands } = yield* Effect.promise(() => import("../../../../../core/src/cli/oclif/compiled-commands.ts"));
const registry = mcpRegistryFromCompiled({ "app:config": compiledCommands["app:config"] });
const catalog = buildCatalog({
  commandEntries: registry.commandEntries,
  effective: computeEffectiveAllowlist({ defaults: ["app:config:get", "app:config:view"] }),
});
const frames = [
  { jsonrpc: "2.0", id: 1, method: "initialize", params: {} },
  { jsonrpc: "2.0", method: "notifications/initialized" },
  { jsonrpc: "2.0", id: 2, method: "tools/list", params: {} },
  {
    jsonrpc: "2.0",
    id: 3,
    method: "tools/call",
    params: { name: "app:config:get", arguments: { args: { key: "name" } } },
  },
];
const input = new ReadableStream<Uint8Array>({
  start: (controller) => {
    controller.enqueue(new TextEncoder().encode(frames.map((frame) => JSON.stringify(frame)).join("\n") + "\n"));
    controller.close();
  },
});
const writes: string[] = [];
const transport = yield* makeStdioMcpTransport({
  catalog,
  input,
  write: (line) => Effect.sync(() => writes.push(line)),
});
const incoming = yield* transport.receive;
expect(incoming._tag).toBe("Some");
if (incoming._tag !== "Some") return yield* Effect.die(new Error("expected a tool call"));
expect(incoming.value.request).toEqual({
  toolId: "app:config:get",
  input: { args: { key: "name" } },
});
yield* transport.reply({
  id: incoming.value.id,
  ok: true,
  result: {
    envelope: {
      apiVersion: "v4",
      command: "app:config:get",
      ok: true,
      result: { key: "name", value: "mcp-guide" },
    },
    ok: true,
  },
});
while (writes.length < 3) yield* Effect.sleep("10 millis");
const responses = writes.map((line) => JSON.parse(line));
expect(responses).toContainEqual(expect.objectContaining({
  id: 1,
  result: expect.objectContaining({ protocolVersion: "2024-11-05" }),
}));
expect(responses).toContainEqual(expect.objectContaining({
  id: 2,
  result: { tools: expect.arrayContaining([
    expect.objectContaining({ name: "app:config:get" }),
    expect.objectContaining({ name: "app:config:view" }),
  ]) },
}));
expect(responses).toContainEqual(expect.objectContaining({
  id: 3,
  result: expect.objectContaining({ isError: false }),
}));`}
        displayCode={`const child = Bun.spawn(["lando", "mcp"], {
  stdin: "pipe",
  stdout: "pipe",
  stderr: "pipe",
});

const frames = [
  { jsonrpc: "2.0", id: 1, method: "initialize", params: {} },
  { jsonrpc: "2.0", method: "notifications/initialized" },
  { jsonrpc: "2.0", id: 2, method: "tools/list", params: {} },
  {
    jsonrpc: "2.0",
    id: 3,
    method: "tools/call",
    params: {
      name: "app:config:get",
      arguments: { args: { key: "name" } },
    },
  },
];

for (const frame of frames) child.stdin.write(JSON.stringify(frame) + "\n");
await child.stdin.flush();

const reader = child.stdout.getReader();
const decoder = new TextDecoder();
let buffered = "";
while (true) {
  const chunk = await reader.read();
  if (chunk.done) break;
  buffered += decoder.decode(chunk.value, { stream: true });
  const lines = buffered.split("\n");
  buffered = lines.pop() ?? "";
  if (lines.some((line) => JSON.parse(line).id === 3)) break;
}

await child.stdin.end();
await child.exited;`}
      />

      <Variable name="unusable" value="missing or unsupported stdin/stdout descriptor" display="Serve mode rejects unavailable descriptors and character devices" />
      <Variable name="machineOutput" value="lando mcp --format=json" display="Serve mode rejects command-result machine output; use `lando mcp --list --format=json` instead" />
      <Run
        runtime="library"
        code={`const { classifyMcpServeStartup } = yield* Effect.promise(() => import("../../../../../core/src/mcp/stdio-limits.ts"));
const pipe = { available: true, tty: false, kind: "fifo" } as const;
expect(classifyMcpServeStartup({ resultFormat: "text", stdin: pipe, stdout: pipe })).toBeUndefined();
const tty = classifyMcpServeStartup({
  resultFormat: "text",
  stdin: { available: true, tty: true, kind: "character" },
  stdout: pipe,
});
const unavailable = classifyMcpServeStartup({
  resultFormat: "text",
  stdin: pipe,
  stdout: { available: false, tty: false, kind: "other" },
});
const machine = classifyMcpServeStartup({ resultFormat: "json", stdin: pipe, stdout: pipe });
expect(tty?._tag).toBe("McpTransportError");
expect(unavailable?._tag).toBe("McpTransportError");
expect(machine?._tag).toBe("McpTransportError");
expect(tty?.message).toBe("MCP serve mode requires usable non-TTY stdin and stdout descriptors.");
expect(unavailable?.remediation).toContain("stdin and stdout connected as pipes");
expect(machine?.message).toBe("MCP serve mode cannot use command-result machine output.");
expect(machine?.remediation).toContain("add --list");`}
        displayCode={`// Let the MCP client own both protocol pipes.
Bun.spawn(["lando", "mcp"], { stdin: "pipe", stdout: "pipe", stderr: "pipe" });

// Do not start serve mode directly in a terminal or with --format=json.
// Audit the terminating, machine-readable catalog separately:
Bun.spawn(["lando", "mcp", "--list", "--format=json"]);`}
      />
    </Step>
  </Scenario>

  <Scenario id="bounded-read-only-regressions" render>

      <Variable name="outbound" value="1024 messages / 8 MiB / 5 seconds" display="The outbound queue, queued bytes, and blocked-write duration are bounded" />
      <Run
        runtime="library"
        code={`const { makeStdioMcpTransport } = yield* Effect.promise(() => import("../../../../../core/src/mcp/stdio-transport.ts"));
const { MAX_OUTSTANDING_REQUESTS } = yield* Effect.promise(() => import("../../../../../core/src/mcp/stdio-limits.ts"));
const { buildCatalog, computeEffectiveAllowlist } = yield* Effect.promise(() => import("../../../../../core/src/mcp/catalog.ts"));
const { mcpRegistryFromCompiled } = yield* Effect.promise(() => import("../../../../../core/src/cli/commands/meta/mcp.ts"));
const { default: compiledCommands } = yield* Effect.promise(() => import("../../../../../core/src/cli/oclif/compiled-commands.ts"));
const registry = mcpRegistryFromCompiled({ "app:config": compiledCommands["app:config"] });
const catalog = buildCatalog({
  commandEntries: registry.commandEntries,
  effective: computeEffectiveAllowlist({ defaults: ["app:config:get", "app:config:view"] }),
});
expect(catalog.tools.map((tool) => tool.toolId)).toEqual(["app:config:get", "app:config:view"]);
expect(catalog.tools.find((tool) => tool.toolId === "app:config:get")?.inputSchema).toMatchObject({
  properties: { args: { required: ["key"] }, flags: { properties: {} } },
});
expect(catalog.tools.find((tool) => tool.toolId === "app:config:view")?.inputSchema).toMatchObject({
  properties: { args: { properties: {} }, flags: { properties: {} } },
});
const frames = Array.from({ length: MAX_OUTSTANDING_REQUESTS + 1 }, (_, index) => ({
  jsonrpc: "2.0",
  id: index + 1,
  method: "tools/call",
  params: { name: "app:config:get", arguments: { args: { key: "name" } } },
}));
const input = new ReadableStream<Uint8Array>({
  start: (controller) => {
    controller.enqueue(new TextEncoder().encode(frames.map((frame) => JSON.stringify(frame)).join("\n") + "\n"));
    controller.close();
  },
});
const writes: string[] = [];
const transport = yield* makeStdioMcpTransport({
  catalog,
  input,
  write: (line) => Effect.sync(() => writes.push(line)),
});
for (let index = 0; index < MAX_OUTSTANDING_REQUESTS; index += 1) {
  const incoming = yield* transport.receive;
  expect(incoming._tag).toBe("Some");
}
while (writes.length < 1) yield* Effect.sleep("10 millis");
expect(JSON.parse(writes[0] ?? "{}")).toMatchObject({
  jsonrpc: "2.0",
  id: MAX_OUTSTANDING_REQUESTS + 1,
  error: { code: -32000, message: "Server busy" },
});`}
        displayCode={`// Keep protocol input and output bounded. Lando rejects oversized or stalled
// peers, caps unresolved calls at 256, and returns Server busy for request 257.
// The app-config projection is intentionally read-only:
const appConfigTools = ["app:config:get", "app:config:view"];
console.log(appConfigTools);`}
      />
    </Step>
  </Scenario>
